import socket

# TCP server details
server_ip = '192.168.10.1'
server_port = 9999

# Request data
request_data = {
    'STX': 0x02,
    'Command': 'YourCommand',
    'CmdType': 'YourCmdType',
    'Acknowledge': 'YourAcknowledge',
    'Reserve': 'YourReserve',
    'ETX': 0x03
}

# Convert the request data to bytes
request_bytes = bytes([request_data['STX']]) + \
               request_data['Command'].encode('ascii') + \
               request_data['CmdType'].encode('ascii') + \
               request_data['Acknowledge'].encode('ascii') + \
               request_data['Reserve'].encode('ascii') + \
               bytes([request_data['ETX']])

# Create a TCP socket and connect to the server
client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
client_socket.connect((server_ip, server_port))

# Send the request
client_socket.sendall(request_bytes)

# Receive the response
response = client_socket.recv(1024)

# Print the response
print(response)

# Close the socket
client_socket.close()
